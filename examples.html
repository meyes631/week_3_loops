Examples

Circles inside circles

function setup() { createCanvas(400, 400); background("black"); colorMode(HSB, 100); noFill(); for(var i = 0; i < 20; i++){ stroke(i*7,100,100); ellipse(200, 200, i * 30, i * 30); } }

Circles inside circles with Arrays

var sizes = [100,50,26,13,87]; function setup() { createCanvas(400, 400); background("black"); stroke("white"); noFill(); for(var x = 0; x < sizes.length; x++){ ellipse(200, 200, sizes[x], sizes[x]); } }

Nested Loops with if statements

function setup() { createCanvas(400, 400); background(0, 0, 0); for (var y = 0; y < 4; y++) { for (var x = 0; x < 6; x++) { rect(x * 100+10, y * 100+10, 70, 70); } } }

Arrays and loops

var myArray = [ ["circle","line","line","line"], ["line","circle","line","line"], ["line","line","circle","line"], ["line","line","line","circle"], ]; function setup() { createCanvas(400, 400); background("black"); stroke("white"); for(var row = 0; row < myArray.length; row++){ for(var column = 0; column < myArray[row].length;column++){ if (myArray[row][column] == "circle"){ ellipse(row*100, column*100, 100, 100); } else if (myArray[row][column] == "line") line(row*100, column*100, row*100+100 , column*100+100); } } }

Follow the mouse

function setup() { createCanvas(windowWidth, windowHeight); } function draw() { background("lightblue"); fill("black"); ellipse(mouseX, mouseY, 100, 100); }

Change when the mouse clicks

function setup() { createCanvas(windowWidth, windowHeight); } function draw() { if (mouseIsPressed) { background(255); fill(0, 255, 0); ellipse(width * 0.5, height * 0.5, 200, 200); } else { background(0); fill(0, 0, 255); ellipse(width * 0.5, height * 0.5, 200, 200); } }

Animation

var Yposition = 0; function setup() { createCanvas(windowWidth, windowHeight); } function draw() { Yposition = Yposition + 1; if (Yposition > windowHeight) { Yposition = 0; } background("lightblue"); ellipse(200, Yposition, 50, 50); }
